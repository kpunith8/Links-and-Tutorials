# Using CDN with a fallback:

	- 	<script> window.jQuery || document.write('<script src="jquery.min.js"><\/script>') </script>

# Selectors:

	- 	selector caching: store the selector it to a variable if it is used multiple times to avoid re-querying the DOM
	-	var panels = $(".panel", container); // search for panel classes in container (Or) var panels = container.find(".panel");

# Creating custom pseudo-class selector:

	can be created using $.expr, for ex:

	$.extend($.expr[":"], {
		hasArialFont: function(element) {
		return $(element).css("font-family") === "Arial";
		}
	});

	Usage:

	$("div:hasArialFont").click(function() {

	});
